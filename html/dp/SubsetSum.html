Subset Sum

boolean[][] b;

public boolean getSubsetSumUsingBottomUp(int[] arr, int sum, int n) {

              b = new boolean[n+1][sum+1];      

              for(int i=0; i<=n; i++) {

                     for(int j=0; j<=sum; j++) {

                           if(j==0)

                                  b[i][j] = true;

                           else if(i==0)

                                  b[i][j] = false;

                     }

              }

               for(int i=1; i<=n; i++) {

                     for(int j=1; j<=sum; j++) {

                           if(arr[i-1]<=j) {

                                  b[i][j] =  b[i-1] [j-arr[i-1]] || b[i-1][j];

                           }else

                                  b[i][j]= b[i-1][j];

                     }

              }            

              return b[n][sum];

       }
